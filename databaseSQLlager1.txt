CREATE TABLE Lager (
LNr INT AUTO_INCREMENT,
Gateadresse VARCHAR(50),
PostNr INT(4),
KvmPris INT NOT NULL,
PRIMARY KEY (LNr)
);

CREATE TABLE LagerRom (
LNr INT NOT NULL,
RNr INT NOT NULL,
Kvm INT NOT NULL,
PRIMARY KEY (LNr,RNr),
FOREIGN KEY (LNr) REFERENCES Lager(LNr)
);

CREATE TABLE Kunde (
KNr INT AUTO_INCREMENT,
Fornavn VARCHAR (50),
Etternavn VARCHAR (50),
Tlf INT(8),
PRIMARY KEY (KNr)
);

CREATE TABLE Avtale (
ANr INT AUTO_INCREMENT,
KNr INT NOT NULL,
LNr INT NOT NULL,
RNr INT NOT NULL,
FraDato DATE NOT NULL,
TilDato DATE,
PRIMARY KEY (ANr),
FOREIGN KEY (KNr) REFERENCES Kunde(KNr),
FOREIGN KEY (LNr) REFERENCES Lager(LNr),
FOREIGN KEY (LNr,RNr) REFERENCES LagerRom(LNr,RNr)
);
